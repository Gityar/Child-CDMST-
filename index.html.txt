    function setLanguage(lang) {
        currentLang = lang;
        document.getElementById('en-btn').classList.toggle('active', lang === 'en');
        document.getElementById('am-btn').classList.toggle('active', lang === 'am');
        
        document.getElementById('app-title').innerText = uiText[lang].title;
        document.getElementById('app-subtitle').innerText = uiText[lang].subtitle;
        document.getElementById('input-prompt').innerText = uiText[lang].prompt;
        document.getElementById('btn-text').innerText = uiText[lang].btn;
        document.getElementById('result-title').innerText = uiText[lang].res;
        
        if(document.getElementById('results').style.display === 'block') showMilestones();
    }

    function showMilestones() {
        const age = parseInt(document.getElementById('ageInput').value);
        if (isNaN(age) || age < 0 || age > 59) {
            alert(currentLang === 'en' ? "Please enter age between 0-59" : "እባክዎን ከ 0-59 መካከል ያስገቡ");
            return;
        }

        // Find the closest milestone age without going over
        let selectedData = milestoneData[0];
        for (let data of milestoneData) {
            if (age >= data.age) {
                selectedData = data;
            }
        }

        document.getElementById('age-display').innerText = uiText[currentLang].ageText + age + uiText[currentLang].months;

        const content = document.getElementById('milestone-content');
        content.innerHTML = '';
        
        uiText[currentLang].domains.forEach((domain, index) => {
            content.innerHTML += `
                <div class="domain-card">
                    <span class="domain-name">${domain}</span>
                    <div>${selectedData[currentLang][index]}</div>
                </div>
            `;
        });

        document.getElementById('counseling-content').innerHTML = `
            <h3>${uiText[currentLang].tips}</h3>
            <p>${selectedData.tips[currentLang]}</p>
        `;

        document.getElementById('results').style.display = 'block';
        window.scrollTo(0, document.body.scrollHeight);
    }
